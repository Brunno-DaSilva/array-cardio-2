Start with the following Arrays

          
          const people = [
            { name: "Bruno", year: 1989 },
            { name: "Liz", year: 1991 },
            { name: "Ian", year: 2017 },

          const comments = [
            { text: "Let's Code an hour a day", id: 33325 },

Is at least one person 19 or older?

          
          /*******************************
          *           some()
          *******************************/

          // Array.prototype.some()
           
          const isAdult = people.some(person => {
            const currentYear = new Date().getFullYear();
            return currentYear - person.year >= 19;
          });
          console.table(isAdult);

Is everyone 19 or older?

          
            /*******************************
            *         every()
            *******************************/
            // Array.prototype.every()

            const areTheyAdults = people.every(person => {
              const currentYear = new Date().getFullYear();
              return currentYear - person.year <= 19;
            });
            console.log(areTheyAdults);

Find the comment with the ID of 33325

          
            /*******************************
            *           find()
            *******************************/
            // Array.prototype.find()
           // Find is like filter, but instead returns just the one you are 
          // looking to find a specific instance

          const comment = comments.find(comment => comment.id === 33325);
          console.log(comment);
